const SHEET_ID = "1MIf07luEqKvj3IHU4-4GAeL7RrhUuksUbsLg5s-0_P0"; const SHEET_NAME = "Sheet1"; const DELETE_PASSWORD = "admin123"; const HEADERS = [ "No", "Company Name", "Location", "Contact Person", "Email", "Contact Details", "Interests", "Status", "G.M. Actions", "G.M. Contact Person", "Notes" ]; function doGet(e) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); const values = sheet.getDataRange().getValues(); const rows = values.slice(1).filter(row => row.some(cell => cell !== "")); const data = rows.map(row => ({ no: row[0], company: row[1], location: row[2], contactPerson: row[3], email: row[4], phone: row[5], interests: row[6], status: row[7], action: row[8], gmOwner: row[9], notes: row[10] })); return jsonResponse(data, e && e.parameter && e.parameter.callback); } function doPost(e) { const body = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); if (body.type === "add") { const nextNo = getNextNo(sheet); const record = body.record || {}; sheet.appendRow([ nextNo, record.company || "", record.location || "", record.contactPerson || "", record.email || "", record.phone || "", record.interests || "", record.status || "New Lead", record.action || "", record.gmOwner || "", record.notes || "" ]); return jsonResponse({ success: true, no: nextNo }); } if (body.type === "update") { const row = findRowByNo(sheet, body.no); if (!row) return jsonResponse({ success: false, error: "Record not found" }); const column = HEADERS.indexOf(body.field) + 1; if (column < 1) return jsonResponse({ success: false, error: "Invalid field" }); sheet.getRange(row, column).setValue(body.value || ""); return jsonResponse({ success: true }); } if (body.type === "delete") { if (body.password !== DELETE_PASSWORD) { return jsonResponse({ success: false, error: "Incorrect delete password" }); } const row = findRowByNo(sheet, body.no); if (!row) return jsonResponse({ success: false, error: "Record not found" }); sheet.deleteRow(row); return jsonResponse({ success: true }); } return jsonResponse({ success: false, error: "Unknown request" }); } function getNextNo(sheet) { const lastRow = sheet.getLastRow(); if (lastRow < 2) return 1; const values = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const numbers = values.map(Number).filter(num => !isNaN(num)); return numbers.length ? Math.max(...numbers) + 1 : 1; } function findRowByNo(sheet, no) { const lastRow = sheet.getLastRow(); if (lastRow < 2) return null; const values = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const index = values.findIndex(value => String(value) === String(no)); return index === -1 ? null : index + 2; } function jsonResponse(data, callback) { const json = JSON.stringify(data); if (callback) { return ContentService .createTextOutput(`${callback}(${json});`) .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
